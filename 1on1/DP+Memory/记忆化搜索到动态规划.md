class Solution:
    def climbStairs(self, n: int) -> int:
        if not n:
            return 0
        memo = [-1 for _ in range(n + 1)]
        self.helper(n, memo)
        return memo[n]
    
    def helper(self, n, memo):
        if memo[n] != -1:
            return
        if n == 0 or n == 1:
            memo[n] = 1
            return
        self.helper(n - 1, memo)
        self.helper(n - 2, memo)
        memo[n] = memo[n - 1] + memo[n - 2]

            self.dfs(i - 2, s, memo)

class Solution:
    def numDecodings(self, s: str) -> int:
        if not s:
            return 0
        n = len(s)
        f = [0 for _ in range(n + 1)]
        f[0] = 1
        if s[0] == '0':
            f[1] = 0
        else:
            f[1] = 1
        
        for i in range(2, n + 1):
            num = int(s[i - 1])
            if num >=1 and num <= 9:
                f[i] = f[i - 1]
            two_num = int(s[i - 2 : i])
            if two_num >= 10 and two_num <= 26:
                f[i] += f[i - 2]
        return f[n]

# 动态规划
class Solution:
    def getNumberOfWays(self, n, m, limit, cost):
        f = [[0 for i in range(m + 1)] for i in range(n + 1)]
        f[0][m] = 1
        for i in range(n + 1):
            for j in range(m + 1):
                for t in range(max(0, i - limit), i):
                    if j + cost[i] <= m:
                        f[i][j] += f[t][j + cost[i]]
        ans = 0
        for i in range(m + 1):
            ans += f[n][i]
        return ans
# 记忆化搜索
class Solution:
    """
    @param n: the max identifier of planet.
    @param m: gold coins that Sven has.
    @param limit: the max difference.
    @param cost: the number of gold coins that reaching the planet j through the portal costs.
    @return: return the number of ways he can reach the planet n through the portal.
    """
    def getNumberOfWays(self, n, m, limit, cost):
        f = [[-1 for j in range(m + 1)] for i in range(n + 1)]
        ans = 0
        for j in range(m + 1):
            self.dfs(n, j, f, n, m, limit, cost)
            ans += f[n][j]
        return ans
        
    def dfs(self, i, j, f, n, m, limit, cost):
        if f[i][j] != -1:
            return;
        
        # 位于0号星球的时候，金币必须为m，因为没有花费
        if i == 0:
            if j == m:
                f[i][j] = 1
            else:
                f[i][j] = 0
            return
        # 初始化
        f[i][j] = 0
        if j + cost[i] > m:
            return
        for k in range(i - limit, i):
            if k < 0:
                continue
            # 递归搜索到达星球k，剩余j + cost[i]的状态有多少种方式
            self.dfs(k, j + cost[i], f, n, m, limit, cost)
            f[i][j] += f[k][j + cost[i]]
